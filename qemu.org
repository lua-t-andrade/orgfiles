#+title: INSTRUCTIONS FOR QEMU USE

** Creates a disk image of name and X size
#+begin_src bash
  qemu-img create -f qcow2 name.qcow2 XG
#+end_src

** Boots system with cdrom and uses disk image as storage (hard drive)
#+begin_src bash
  qemu-system-x86_64 -m X -nic user -boot once=d -cdrom X.iso -drive file=name.qcow2 -display gtk -enable-kvm
#+end_src

** Boots system with no cdrom
#+begin_src bash 
  qemu-system-x86_64 -m X -nic user -drive file=name.qcow2
#+end_src
